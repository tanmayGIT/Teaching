\begin{frame}
\frametitle{Connaissances et compétences}
\footnotesize
\only<1>{
\begin{block}{Connaissances}
\begin{itemize}
\item Architecture de Von~Neumann;
\item Représentation des données en mémoire;
\item Logique des calculs (UAL, circuits logiques);
\item Algèbre de Boole (De Morgan);
\item Entrées/Sorties;
\item Mécanisme d'interruption.
\end{itemize}
\end{block}

\begin{block}{Compétences}
\begin{itemize}
\item Administrer un système;
\item Comprendre la logique d'exécution d'une application;
\item Développer des applications utilisant des appels systèmes;
\item Utiliser/créer des librairies.
\end{itemize}
\end{block}
\normalsize
}
